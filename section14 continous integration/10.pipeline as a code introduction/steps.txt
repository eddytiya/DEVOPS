"#we are going to write some sode which does 3 things fetch the code build the code and run some tests
#first pipeline code

pipeline {
    agent any
    tools {
        maven "MAVEN3"
        jdk "OracleJDK11"
    }

    stages {
        stage('fetch code') {
            steps {
                git branch: 'main', url: 'https://github.com/hkhcoder/vprofile-project.git'
            }
        }
        stage('Build') {
            steps {
                sh 'mvn install -DskipTests'
            }
            post {
                success {
                    echo "Archiving artifacts now."
                    archiveArtifacts artifacts: '**/*.war'
                }
            }
        }
        stage('UNIT TESTS') {
            steps {
                sh 'mvn test'
            }
        }
    }
}

"
